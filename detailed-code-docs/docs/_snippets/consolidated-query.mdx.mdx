&lt;ResponseField name="requestId" type="string" required&gt;
  The ID of the request.
&lt;/ResponseField&gt;
&lt;ResponseField name="status" type="string" required&gt;
  The status of the request. Can be one of `processing`, `completed`, or `failed`.
&lt;/ResponseField&gt;
&lt;ResponseField name="startedAt" type="string" optional&gt;
  The date and time the request was started - formatted `YYYY-MM-DDTHH:MM:SSZ` as per ISO 8601.
&lt;/ResponseField&gt;
&lt;ResponseField name="resources" type="string[]" optional&gt;
  Comma-separated list of resource types.
&lt;/ResponseField&gt;
&lt;ResponseField name="dateFrom" type="string" optional&gt;
  The start date (inclusive) for which to filter resources to count - formatted `YYYY-MM-DD` as per
  ISO 8601.
&lt;/ResponseField&gt;
&lt;ResponseField name="dateFrom" type="string" optional&gt;
  The end date (inclusive) for which to filter resources to count - formatted `YYYY-MM-DD` as per
  ISO 8601.
&lt;/ResponseField&gt;
&lt;ResponseField name="conversionType" type="string" optional&gt;
  The format of the resulting data. Can be one of `pdf`, `html`, or `json`.
&lt;/ResponseField&gt;
