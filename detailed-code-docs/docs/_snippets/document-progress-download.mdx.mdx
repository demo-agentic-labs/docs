&lt;ResponseField name="download" type="Progress"&gt;
  &lt;Expandable title="Download Properties"&gt;
    &lt;ResponseField name="status" type="string" required&gt;
      The status of querying document references across HIEs, either `processing` or `completed`.
    &lt;/ResponseField&gt;

    &lt;ResponseField name="total" type="number"&gt;
      The total number of documents to be queried.
    &lt;/ResponseField&gt;

    &lt;ResponseField name="successful" type="number"&gt;
      The number of documents successfully downloaded.
    &lt;/ResponseField&gt;

    &lt;ResponseField name="errors" type="number"&gt;
      The number of documents that failed to download.
    &lt;/ResponseField&gt;

  &lt;/Expandable&gt;
&lt;/ResponseField&gt;
