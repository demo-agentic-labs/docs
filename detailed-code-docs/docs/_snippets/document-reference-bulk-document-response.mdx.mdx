&lt;ResponseField name="id" type="string" required&gt;
  The Document ID.
&lt;/ResponseField&gt;

&lt;ResponseField name="url" type="string" required&gt;
  The URL for downloading the document.
&lt;/ResponseField&gt;

&lt;ResponseField name="fileName" type="string" required&gt;
  Name of the file.
&lt;/ResponseField&gt;

&lt;ResponseField name="description" type="string"&gt;
  Human-readable description of the Document.
&lt;/ResponseField&gt;

&lt;ResponseField name="status" type="string"&gt;
  Can be one of `current`, `superceded`, or `entered in error`.
&lt;/ResponseField&gt;

&lt;ResponseField name="indexed" type="string"&gt;
  When this Document reference was created - formatted to ISO 8601.
&lt;/ResponseField&gt;

&lt;ResponseField name="mimeType" type="string"&gt;
    Document MIME type - usually `application/xml`, or `application/pdf`.
&lt;/ResponseField&gt;

&lt;ResponseField name="size" type="string"&gt;
  Size of the Document in bytes.
&lt;/ResponseField&gt;

&lt;ResponseField name="type" type="CodableConcept"&gt;
    What kind of document this is (LOINC if possible).
    &lt;Expandable title="CodableConcept properties"&gt;
        &lt;ResponseField name="coding" type="Coding[]"&gt;
            Array containing the coding defined by a terminology system.
            &lt;Expandable title="Coding properties"&gt;
                &lt;ResponseField name="system" type="string"&gt;
                    Identity of the terminology system.
                &lt;/ResponseField&gt;
                &lt;ResponseField name="code" type="string"&gt;
                    Symbol in syntax defined by the system.
                &lt;/ResponseField&gt;
                &lt;ResponseField name="display" type="string"&gt;
                    Representation defined by the system.
                &lt;/ResponseField&gt;
            &lt;/Expandable&gt;
        &lt;/ResponseField&gt;
        &lt;ResponseField name="text" type="string"&gt;
            Plain text representation of the concept.
        &lt;/ResponseField&gt;
    &lt;/Expandable&gt;

&lt;/ResponseField&gt;
