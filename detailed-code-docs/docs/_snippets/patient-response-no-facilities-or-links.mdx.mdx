&lt;ResponseField name="id" type="string" required&gt;
  The ID assigned to this Patient. This ID will be used to uniquely identify this Patient in medical
  documents.
&lt;/ResponseField&gt;

&lt;ResponseField name="externalId" type="string"&gt;
  An external Patient ID to associate to a Patient in Metriport.
&lt;/ResponseField&gt;

&lt;ResponseField name="firstName" type="string[]" required&gt;
  The Patient's first name(s).
&lt;/ResponseField&gt;

&lt;ResponseField name="lastName" type="string[]" required&gt;
  The Patient's last name(s).
&lt;/ResponseField&gt;

&lt;ResponseField name="dob" type="string" required&gt;
  The Patient's date of birth (DOB), formatted `YYYY-MM-DD` as per ISO 8601.
&lt;/ResponseField&gt;

&lt;ResponseField name="genderAtBirth" type="string" required&gt;
  The Patient's gender at birth, can be one of `M` or `F` or `O` or `U`. Use `O` (other) when the patient's gender is known but it is not `M` or `F`, i.e intersex or hermaphroditic. 
  Use U (unknown) when the patient's gender is not known.
&lt;/ResponseField&gt;

&lt;ResponseField name="personalIdentifiers" type="PersonalIdentifier[]" optional&gt;
  An array of the Patient's personal IDs, such as a driver's license. May be empty.

  &lt;Expandable title="PersonalIdentifier properties"&gt;
    &lt;ResponseField name="type" type="string" required&gt;
      The ID type - currently `driversLicense` is supported.
    &lt;/ResponseField&gt;

    &lt;ResponseField name="state" type="string" required&gt;
      The 2 letter state acronym where this ID was issued, for example: `CA`.
    &lt;/ResponseField&gt;

    &lt;ResponseField name="value" type="string" required&gt;
      The ID number.
    &lt;/ResponseField&gt;

  &lt;/Expandable&gt;
&lt;/ResponseField&gt;

&lt;ResponseField name="address" type="Address[]" required&gt;
  &lt;Snippet file="patient-address-response.mdx" /&gt;
&lt;/ResponseField&gt;

&lt;ResponseField name="contact" type="Contact[]"&gt;
  &lt;Expandable title="properties"&gt;
    &lt;ResponseField name="phone" type="string"&gt;
      The Patient's 10 digit phone number, formatted `1234567899`.
    &lt;/ResponseField&gt;

    &lt;ResponseField name="email" type="string"&gt;
      The Patient's email address.
    &lt;/ResponseField&gt;

  &lt;/Expandable&gt;
&lt;/ResponseField&gt;
